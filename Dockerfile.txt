# Stage 1: Build the React Application (Vite)
FROM node:20-alpine AS builder

WORKDIR /app

# Argumentos de construcción para inyectar variables en el build de Vite
# Vite requiere las variables de entorno en tiempo de compilación
ARG API_KEY
ENV API_KEY=$API_KEY

COPY package*.json ./
RUN npm install

COPY . .
RUN npm run build

# Stage 2: Production Server (Node.js Express)
FROM node:20-alpine

WORKDIR /app

# Configuración de zona horaria para México
RUN apk add --no-cache tzdata
ENV TZ=America/Mexico_City
ENV NODE_ENV=production
ENV PORT=3000

# API Key disponible para el backend en tiempo de ejecución (Gemini SDK)
ARG API_KEY
ENV API_KEY=$API_KEY

# Directorios de persistencia para archivos subidos (logos, evidencias)
RUN mkdir -p /app/uploads && chown -R node:node /app/uploads

# Instalación de dependencias de producción únicamente para reducir tamaño de imagen
COPY --chown=node:node package*.json ./
RUN npm install --omit=dev

# Copiar el código del servidor y el build estático del frontend
COPY --chown=node:node server ./server
COPY --chown=node:node --from=builder /app/dist ./dist

# Puerto de la aplicación
EXPOSE 3000

# Verificación de salud del contenedor apuntando al endpoint de salud del ERP
HEALTHCHECK --interval=30s --timeout=10s --start-period=20s --retries=3 \
  CMD node -e "fetch('http://localhost:3000/api/health').then((r) => r.ok ? process.exit(0) : process.exit(1)).catch(() => process.exit(1))"

# Seguridad: Ejecución como usuario no privilegiado (node)
USER node

# Comando de inicio del servidor maestro
CMD ["node", "server/index.js"]
