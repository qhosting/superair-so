# Stage 1: Build the React Application (Vite)
FROM node:20-alpine AS builder

WORKDIR /app

# Argumentos de construcción para inyectar variables en el build de Vite
ARG API_KEY
ENV API_KEY=$API_KEY

# Configuración de resiliencia para NPM - Etapa de Construcción
RUN npm config set fetch-retries 10 && \
    npm config set fetch-retry-mintimeout 30000 && \
    npm config set fetch-retry-maxtimeout 180000 && \
    npm config set fetch-timeout 600000 && \
    npm config set network-timeout 600000 && \
    npm config set maxsockets 5 && \
    npm config set loglevel warn

# Instalación de dependencias de construcción
COPY package*.json ./
RUN npm install

# Copia de código fuente y construcción del bundle estático
COPY . .
RUN npm run build

# Stage 2: Production Server (Node.js Express)
FROM node:20-alpine

WORKDIR /app

# Configuración de zona horaria para México
RUN apk add --no-cache tzdata
ENV TZ=America/Mexico_City
ENV NODE_ENV=production
ENV PORT=3000

# API Key disponible para el SDK de Gemini en el backend
ARG API_KEY
ENV API_KEY=$API_KEY

# Directorios de persistencia para uploads (evidencias, logos)
RUN mkdir -p /app/uploads && chown -R node:node /app/uploads

# Configuración de resiliencia para NPM - Etapa de Producción (Evita ECONNRESET aquí también)
RUN npm config set fetch-retries 10 && \
    npm config set fetch-retry-mintimeout 30000 && \
    npm config set fetch-retry-maxtimeout 180000 && \
    npm config set fetch-timeout 600000 && \
    npm config set network-timeout 600000 && \
    npm config set maxsockets 3 && \
    npm config set loglevel warn

# Instalación de dependencias de producción únicamente
COPY --chown=node:node package*.json ./
RUN npm install --omit=dev

# Copiar lógica del servidor y el build estático generado en la etapa 1
COPY --chown=node:node server ./server
COPY --chown=node:node --from=builder /app/dist ./dist

# Puerto expuesto
EXPOSE 3000

# Verificación de salud apuntando al endpoint interno del ERP
HEALTHCHECK --interval=30s --timeout=10s --start-period=20s --retries=3 \
  CMD node -e "fetch('http://localhost:3000/api/health').then((r) => r.ok ? process.exit(0) : process.exit(1)).catch(() => process.exit(1))"

# Seguridad: Ejecución como usuario no privilegiado
USER node

# Comando de inicio del servidor maestro Express
CMD ["node", "server/index.js"]
